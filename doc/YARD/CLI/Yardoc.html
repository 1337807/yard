def initialize
  @options = SymbolHash[
    :format => :html, 
    :template => :default, 
    :markup => :rdoc,
    :serializer => YARD::Serializers::FileSystemSerializer.new, 
    :files => [],
    :visibilities => [:public],
    :verifier => nil
  ]
  @files = []
  @reload = true
  @generate = true
  @options_file = DEFAULT_YARDOPTS_FILE
end

def run(*args)
  args += support_rdoc_document_file!
  optparse(*yardopts)
  optparse(*args)
  Registry.load(files, reload)
  
  if generate
    Templates::Engine.generate(all_objects, options)
  end
  
  true
end
