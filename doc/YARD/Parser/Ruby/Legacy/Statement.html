def inspect
  l = line - 1
  to_s.split(/\n/).map do |text|
    "\t#{l += 1}:  #{text}"
  end.join("\n")
end

def to_s(include_block = true)
  tokens.map do |token|
    RubyToken::TkBlockContents === token ? block.to_s : token.text
  end.join
end
