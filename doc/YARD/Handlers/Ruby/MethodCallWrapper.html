def matches?(node)
  case node.type
  when :var_ref
    if !node.parent || node.parent.type == :list
      return true if node[0].type == :ident && node[0][0] == name
    end
  when :fcall, :command
    return true if node[0][0] == name
  when :call, :command_call
    return true if node[2][0] == name
  end
  false
end
