def run(*args)
  Registry.load
  optparse(*args)
  
  contents = objects.map {|o| o.format(options) }.join("\n")
  Templates::Engine.render(:format => :dot, :type => :layout, 
    :serializer => @serializer, :contents => contents) 
end
